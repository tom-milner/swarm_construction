 19    """The lowest level class to use in a SimulationEngine simulation. SimulationObjects support simple physics, and can render themselves to the simulation.
 20    Simulation "agents" should inherit and extend upon this class.
 21
 22    IMPORTANT: any class members (functions/variables) beginning with underscores (_) SHOULD NOT be used by any external or child classes. Whilst this may seem limiting, it makes the simulation more realistic.
 23
 24    SimulationObjects can either move along vectors specified by a direction and speed, or along an orbit.
 25    """

 28        self,
 29        sim_engine,

 32        color: Colour = Colour.white,
 33        direction: float = math.pi,
 34        speed: int = 0,
 35        label: int = None,
 36    ):
 37        """Initialise a SimulationObject to use in a SimulationEngine simulation.
 38
 39        Args:
 40            sim_engine (SimulationEngine): The SimulationEngine to use this SimulationObject in. This simulation object will automatically add itself to the SimulationEngine.
 41            pos (list, optional): Object's position, as an [x,y] vector. Defaults to [0, 0].
 42            radius (int, optional): Radius of the object. Defaults to 30.
 43            color (tuple, optional): Colour of the object, as an RGB vector (0-255,0-255,0-255). Defaults to white (255, 255, 255).
 44            direction (_type_, optional): Direction of the object, IN RADIANS. Defaults to math.pi.
 45            speed (int, optional): Speed of the object, in PIXLES PER SECOND. Defaults to 0.
 46        """
 47
 48        # Initialise private vars - agents can't use these!
 49        self._sim_engine = sim_engine
 50        self._pos = pos
 51        self._radius = int(radius)
 52        self._comms_radius = (
 53            radius * 2 * 4
 54        )  # based on KiloBot communications distance 'Kilobot: A Low Cost Scalable Robot System for Collective Behaviors'
 55        self._direction = direction
 56        self._orbit_object = None
 57        self._orbit_direction = OrbitDirection.CLOCKWISE
 58
 59        # Initialise public vars - agents can access these.
 60        self.color = color
 61        self.speed = speed  # Speed is measured in pixels per second!
 62        self.label = label
 63
 64        # Give the object a unique id.
 65        global _object_id
 66        self.object_id = _object_id
 67        _object_id += 1
 68
 69        # The neighbourhood we're currently in.
 70        self._neighbourhood = [None, None]
 71
 72        # The font to use for writing the labels.
 73        self._font = pg.font.SysFont("Arial", self._radius)  # Font size = agent radius

 75        # Add ourselves to the SimulationEngine. The simulation engine will automatically update and draw us.
 76        # it will also automatically set the neighbourhood.
 77        self._sim_engine._objects.append(self)

 79    def _move_orbit(self, pixels_per_frame: float):
 80        """Move the object in a circular orbit path around self.orbit_object.
 81        The direction of the object points along (tangential to) the orbit path.
 82        Currently, the object orbits with it's edge touching the orbit object.
 83
 84        Args:
 85            pixels_per_frame (float): Speed of the object in pixels per frame.
 86        """
 87
 88        if self._orbit_object is None:
 89            return
 90
 91        # This is implemented weirdly!
 92        # I didn't want to store a "current_orbit_angle" variable in the class,
 93        # so we spin the object's direction, then position it so that it's
 94        # direction is perpendicular to the orbit_object.
 95        # It works, but there's probbably a better way!
 96
 97        # Distance from center of this object to center of orbit object.
 98        dist = self._radius + self._orbit_object._radius
 99
100        # The angular velocity of this orbit.
101        ang_vel = pixels_per_frame / dist
102
103        # Set the rotation direction (clockwise, anticlockswise) of the orbit using self._orbit_direction.
104        orbit_direction = 1 if self._orbit_direction == OrbitDirection.CLOCKWISE else -1
105
106        # Update the objects direction to point along the orbit path (tangent to the orbit path), and clamp between 0-2PI.
107        self._direction = (self._direction - ang_vel * orbit_direction) % (2 * math.pi)
108
109        # Now we can position the object on the orbit path.
110
111        # Calculate the angle perpendicular to our current direction.
112        perp = self._direction + orbit_direction * (math.pi / 2)
113
114        # Position ourselves in the orbit at that angle.
115        orbit_vec = dist * np.array([math.sin(perp), math.cos(perp)])
116        self._pos = np.add(self._orbit_object._pos, orbit_vec)

118    def _move_vector(self, pixels_per_frame: float):
119        """Move the object along the vector specified by self.direction and pixels_per_frame.
120
121        Args:
122            pixels_per_frame (_type_): Speed of the object in pixels per frame.s
123        """
124
125        # Turn speed and direction into a velocity vector.
126        v = pixels_per_frame * np.array(
127            [math.sin(self._direction), math.cos(self._direction)]
128        )
129        # Apply vector to current position.
130        self._pos = np.add(v, self._pos)

132    def update(self, fps: float):
133        """Update the object in current frame. This is called by the simulation engine (Simulation).
134
135        Args:
136            fps (float): The frames per second since the last call.
137        """
138
139        # If there have been 0 frames since the last call we are on the first frame. Nothing to update, so move straight on!
140        if fps == 0:
141            return
142
143        # If we don't have a neighbourhood, assign one to us.
144        if not np.any(self._neighbourhood):
145            self._sim_engine.assign_neighbourhood(self)
146
147        # If we have no speed, no need to recalculate any movement stuff.
148        if self.speed == 0:
149            return
150
151        # Calculate the number of pixels we must move per frame, in order to produce the desired speed.
152        pixels_per_frame = self.speed / fps
153
154        # If we are orbiting an object, continue that orbit.
155        if self._orbit_object:
156            self._move_orbit(pixels_per_frame)
157        else:
158            # Else, just continue along our movement vector.
159            self._move_vector(pixels_per_frame)
160
161        assert (
162            self._pos[0] < self._sim_engine.window_size
163        ), "SimulationObject x-coordinate is outside the world!"
164        assert (
165            self._pos[1] < self._sim_engine.window_size
166        ), "SimulationObject y-coordinate is outside the world!"
167
168        # Get the sim engine to reassign us to a neighbourhood.
169        self._neighbourhood = self._sim_engine.assign_neighbourhood(self)

177        """

179        # Draw a circle around the position coordinate.

190        # Add the label at the centre of the circle
191        if self.label is not None:

193                str(self.label), True, (0, 0, 0)

195            text_rect = text_surface.get_rect(center=self._pos)

203    def check_collision(self, other_object):
204        """Check if we have collided with another SimulationObject.
205        This currently only works with circles!
206
207        Todo:
208            Add bounding box checks first - may increase efficiency?
209
210        Args:
211            other_object (SimulationObject): The object to check for collision.
212
213        Returns:
214            tuple(bool, [int,int]): A tuple containing a boolean indicating whether collision has occured, and a vector describing how much we overlap the collided object.
215        """
216
217        # Get the vector between the objects.
218        diff = np.subtract(self._pos, other_object._pos)
219
220        # We have collided with the object if the distance between the objects is less than the sum of their radii.
221        threshold = self._radius + other_object._radius
222        distance = np.linalg.norm(diff)
223
224        if distance > threshold:
225            # No collision!
226            return (False, np.array([0, 0]))
227
228        # Collision!
229
230        # Normalise the vector between the objects.
231        norm = diff / distance if distance != 0 else 0
232
233        # Calculate the vector where the object edges are touching but there is no overlap.
234        edges_touching = norm * threshold
235
236        # The overlap is the difference between this vector and our current distance vector.
237        overlap = np.subtract(edges_touching, diff)
238
239        return (True, overlap)

241    def set_orbit_object(
242        self, orbit_object, orbit_direction: OrbitDirection = OrbitDirection.CLOCKWISE
243    ):
244        """Make this object orbit around the provided object, in the provided direction.
245        If we are already touching the object, we will orbit from there.
246        If we aren't touching the object, we will reposition ourselves to be tangential to the object in our current position, then orbit from there.
247
248        Args:
249            orbit_object (SimulationObject): SimulationObject to orbit.
250            orbit_direction (OrbitDirection, optional): Direction to orbit the object (clockwise or anticlockwise). Defaults to OrbitDirection.CLOCKWISE.
251        """
252
253        self._orbit_direction = orbit_direction
254
255        # If we're already orbiting this object, return
256        if orbit_object == self._orbit_object:
257            return
258
259        self._orbit_object = orbit_object
260
261        # If we're already touching the edge of the object, change our
262        # direction to start orbiting from where we are.
263        col = self.check_collision(orbit_object)
264        if col[0]:
265            # Vector from ourselves to the orbit object.
266            diff = np.subtract(self._pos, orbit_object._pos)
267
268            # The angle that is perpencidular to the diff vector.
269            perp = math.atan2(-diff[1], diff[0])
270            self._direction = perp
271
272            # NOTE: technically, we need to reposition ourselves if the overlap is too high (as we cannot exist inside another object), but the _move_orbit function handles this already.
273
274        # If we're not touching the object, just position ourselves so that we're
275        # perpendicular to it at our current direction.
276
277        # The _move_orbit function does everything else!
278        self._move_orbit(0)

280    def get_nearest_neighbours(self, n=None):
281        """Get a sorted list of the SimulationObjects nearest to this object in the Simulation, along with their distances.
282
283         Args:
284            n (int): Number of neighbours to return. Default = all neighbours.
285
286        Returns:
287            list(tuple): The nearest SimulationObjects in the simulation, along with their distances: (neighbour, distance)
288                        Contains all Agents within comms distance (3 agent diameters)
289        """
290        if not np.any(self._neighbourhood):
291            self._neighbourhood = self._sim_engine.assign_neighbourhood(self)
292
293        nearby_agents = self._sim_engine.get_nearby_objects(self._neighbourhood)
294        neighbours = []
295        for obj in nearby_agents:
296
297            # Skip ourselves.
298            if self.object_id == obj.object_id:
299                continue
300
301            # Work out the distance from the current object.
302            diff = np.subtract(self._pos, obj._pos)
303            dist_squared = diff[0] ** 2 + diff[1] ** 2
304
305            # only neighbours within comms distance
306            if dist_squared <= self._comms_radius**2:
307                # Store object and distance in neighbours array.
308                entry = [obj, math.sqrt(dist_squared)]
309                neighbours.append(entry)
310
311        if len(neighbours) == 0:
312            return neighbours
313
314        # Turn neighbours into numpy array, and sort based on distance.
315        neighbours = np.array(neighbours)
316        neighbours = neighbours[neighbours[:, 1].argsort()]
317
318        # Only return the closest n objects.
319        if n is not None:
320            neighbours = neighbours[0:n]
321
322        return neighbours

204    def check_collision(self, other_object):
205        """Check if we have collided with another SimulationObject.
206        This currently only works with circles!
207
208        Todo:
209            Add bounding box checks first - may increase efficiency?
210
211        Args:
212            other_object (SimulationObject): The object to check for collision.
213
214        Returns:
215            tuple(bool, [int,int]): A tuple containing a boolean indicating whether collision has occured, and a vector describing how much we overlap the collided object.
216        """
217
218        # Get the vector between the objects.
219        diff = np.subtract(self._pos, other_object._pos)
220
221        # We have collided with the object if the distance between the objects is less than the sum of their radii.
222        threshold = self._radius + other_object._radius
223        distance = np.linalg.norm(diff)
224
225        if distance > threshold:
226            # No collision!
227            return (False, np.array([0, 0]))
228
229        # Collision!
230
231        # Normalise the vector between the objects.
232        norm = diff / distance if distance != 0 else 0
233
234        # Calculate the vector where the object edges are touching but there is no overlap.
235        edges_touching = norm * threshold
236
237        # The overlap is the difference between this vector and our current distance vector.
238        overlap = np.subtract(edges_touching, diff)
239
240        return (True, overlap)

281    def get_nearest_neighbours(self, n=None):
282        """Get a sorted list of the SimulationObjects nearest to this object in the Simulation, along with their distances.
283
284         Args:
285            n (int): Number of neighbours to return. Default = all neighbours.
286
287        Returns:
288            list(tuple): The nearest SimulationObjects in the simulation, along with their distances: (neighbour, distance)
289                        Contains all Agents within comms distance (3 agent diameters)
290        """
291        if not np.any(self._neighbourhood):
292            self._neighbourhood = self._sim_engine.assign_neighbourhood(self)
293
294        nearby_agents = self._sim_engine.get_nearby_objects(self._neighbourhood)
295        neighbours = []
296        for obj in nearby_agents:
297
298            # Skip ourselves.
299            if self.object_id == obj.object_id:
300                continue
301
302            # Work out the distance from the current object.
303            diff = np.subtract(self._pos, obj._pos)
304            dist_squared = diff[0] ** 2 + diff[1] ** 2
305
306            # only neighbours within comms distance
307            if dist_squared <= self._comms_radius**2:
308                # Store object and distance in neighbours array.
309                entry = [obj, math.sqrt(dist_squared)]
310                neighbours.append(entry)
311
312        if len(neighbours) == 0:
313            return neighbours
314

316        neighbours = np.array(neighbours)

318
319        # Only return the closest n objects.
320        if n is not None:
