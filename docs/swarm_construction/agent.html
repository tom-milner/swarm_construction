 97    def start_edge_following(self, fps, neighbours):
 98        """Start edge following by checking for a bunch of conditions
 99
100        If the conditions are passed, the agent is set to orbit the neighbour it is
101        touching with the highest gradient.
102        Args:
103            fps (int): the sims current frame rate"""
104
105        # get the nearest non-localised neighbours
106        neighbours = [
107            n for n in neighbours if n[0].state == AgentState.IDLE or n[0].speed != 0
108        ]
109
110        # If there are no neighbours, start.
111        if len(neighbours) == 0:
112            return True
113
114        # get gradients of neighbours
115        gradients = [neighbour[0].gradient for neighbour in neighbours]
116
117        # there are neighbours with a higher gradient than us
118        if any(
119            gradient is not None and gradient > self.gradient for gradient in gradients
120        ):
121            return False
122
123        # there are neighbours moving
124        if any(n[0].speed > 0 and n[1] < Agent.radius * 2 * 4 for n in neighbours):
125
126            return False
127
128        equal_grad_neighbours = 0
129        max_grad_neighbour = neighbours[0]
130        for neighbour in neighbours:
131            # counts neighbours within touching distance with the same gradient as the agent
132            if (neighbour[1] <= Agent.radius * 2) and (
133                neighbour[0].gradient == self.gradient
134            ):
135                equal_grad_neighbours += 1
136
137            # find the neighbour with the maximum gradient
138            if (neighbour[1] <= Agent.radius * 2) and (
139                neighbour[0].gradient > max_grad_neighbour[0].gradient
140            ):
141                max_grad_neighbour = neighbour
142
143        # if we're touching 2 neighbours with equal gradient to ourselves
144        # it is likely that we will get trapped between them in a cycle
145        # of collisions and changing orbit objects if we tried to start edge following
146        if equal_grad_neighbours > 1:
147            return False
148
149        if max_grad_neighbour[0].speed == 0:
150            self.set_orbit_object(max_grad_neighbour[0])
151
152        return True

154    def follow_edges(self, neighbours):
155        """Move round the edges of the provided neighbours.
156        Works by orbiting an agent until it collides with another, upon which it orbits the collision agent.
157        Requires the agent to be moving.
158        Args:
159            neighbours (list(tuple)): List of tuples of neighbours and their distances (neighbour, distance) sorted by distance.
160        """
161
162        if len(neighbours) == 0:
163            return
164
165        # Check if we've collided with our nearest neighbour.
166        closest = neighbours[0]
167        collision = self.check_collision(closest[0])
168        if not collision[0]:
169            # If we're not touching anything, do nothing.
170            return
171
172        # If we crash into an agent, stop moving for a bit.
173        if closest[0].speed != 0:
174            self.fix_collision(collision)
175            return
176
177        # Orbit around the new neighbour.
178        self.set_orbit_object(closest[0])

180    def localise(self, neighbours):
181        """Localise ourselves using the surrounding agents. Sets local_pos to be our calculated location.
182
183        Args:
184            neighbours (list(tuple)): List of tuples of nearest neighbours (neighbouring agent, distance).
185        """
186
187        # Don't allow seed robots to relocalise.
188        if self.is_seed:
189            return
190
191        # No need to re-localise if we're stationary.
192        # In reality, robots will need to periodically re-localise, as they may be bumped around.
193        if self.speed == 0:
194            return
195
196        # Get the neighbours that are already localised.
197        localised_neighbours = [
198            n
199            for n in neighbours
200            if n[0].state == AgentState.LOCALISED and n[0].local_pos is not None
201        ]
202
203        # If we have < 3 localised neighbours, we can't localise ourselves
204        if len(localised_neighbours) < 3:
205            self.local_pos = None
206            return
207
208        # DEBUG - return the agents actual position as the localised position.
209        # Useful for debugging other parts of the agent that rely on localisation to be working properly.
210        # seed = self._sim_engine._objects[0]
211        # self.local_pos = [self._pos[0] - (seed._pos[0] - seed.local_pos[0]),
212        #                              seed._pos[1] - self._pos[1]]
213        # return
214
215        # Set a starting local_pos if necessary.
216        pos = [-self.radius, +self.radius] if self.local_pos is None else self.local_pos
217
218        # NOTE: This localisation algorithm is a bit rubbish. It's based off the paper, and was designed to manage
219        # the constraints they were facing with small, low-power, low-compute robots, and asynchronous comms.
220
221        # Their robots are constantly localising in the background. We only have one frame.
222
223        last_pos = pos
224        
225        dist = 1
226        threshold = 0.001
227        while dist > threshold**2:
228            for n in localised_neighbours:
229                agent = n[0]
230                measured_dist = n[1]
231
232                # vector from neighbour local_pos to ourselves.
233                neighbour_vec = np.subtract(pos, agent.local_pos)
234
235                # the distance between our local_pos and the neighbours local_pos.
236                calculated_dist = np.linalg.norm(neighbour_vec)
237
238                # unit vector pointing from neighbour to ourselves.
239                v = neighbour_vec / calculated_dist
240
241                # scale unit vector to size of actual distance measured.
242                actual_vec = v * measured_dist
243
244                # compute new position of where we should be, based on the measured distance.
245                new_pos = np.add(agent.local_pos, actual_vec)
246
247                # In the paper, they only move 1/4 of the way to the new position.
248                # move towards new position
249                pos_diff = np.subtract(pos, new_pos) / 4
250                # pos_diff = np.subtract(pos, new_pos)
251
252                pos = np.subtract(pos, pos_diff)
253
254            # If our new position is different from our last position, we haven't minimised,
255            # so run the localisation again.
256            diff_from_last = np.subtract(pos, last_pos)
257            # This would usually be linalg.norm, but we omit the sqrt to improve performance.
258            dist = diff_from_last[0] ** 2 + diff_from_last[1] ** 2
259            last_pos = pos
260
261        # Add localisation noise to simulate real life.
262        noise_threshold = 0.00
263        pos += random.uniform(-noise_threshold, noise_threshold)
264
265        # save our localised position
266        self.local_pos = pos

268    def is_inside_shape(self) -> bool:
269        """Check if our localised position (local_pos) is inside the shape (target_shape)
270
271        Returns:
272            bool: Whether we're in the shape (True) or not (False).
273        """
274        if self.shape is None:
275            return False
276
277        # Can only see if we're inside the shape if we're localised.
278        if self.local_pos is None:
279            return False
280
281        # Map the localised swarm position to the shape, using the bottom left pixel in the shape.
282        # They both use origin (x,y) = (0,0) = bottom left (for now...)
283        size = self.shape.shape_data.size
284        mapped = np.add(self.local_pos, self.shape.bottom_left).round()
285        # print(mapped) # Saved my life during debugging so leaving in as a reminder of what went down here.
286
287        # If we're not in the shapes bounding box, return early.
288        if not (0 < mapped[0] < size[0] - 1) or not (0 < mapped[1] < size[1] - 1):
289            return False
290
291        # We're in the bounding box!
292
293        # Get the raw pixels from the shape, and get the color of the pixel at our mapped position.
294        pixels = self.shape.shape_data.getdata()
295
296        # Calculate 1D index from 2D mapped position.
297        # Pixel origin (0,0) is top left damnit! Flip our mapped position y-axis to deal with this.
298        rows = int(size[0] * (size[1] - mapped[1]))
299        image_idx = int(rows + mapped[0])
300
301        if pixels[image_idx] != self.color_flag:
302            # We are not in our part of the shape :(
303            return False
304
305        # Woohoo! We're in the shape!
306        return True

308    def update_gradient(self, neighbours):
309        """updates the gradient of the agent
310        Works by finding lowest gradient of neighbours and adds 1
311        Args:
312            neighbours (list(tuple)): List of tuples (neighbouring agent, distance)
313        """
314
315        # we only use the closest neighbours (ones we are touching or almost touching)
316        neighbours = [n for n in neighbours if n[1] <= Agent.radius * 2.1]
317
318        # now lets get the gradients
319        gradients = [neighbour[0].gradient for neighbour in neighbours]
320
321        valid_gradients = []
322        # itterate thro and get all valid gradients
323        for i in gradients:
324            # make sure they have a set gradient
325            if i is None:
326                continue
327            valid_gradients.append(i)
328
329        # check we have not got an empty list
330        if not valid_gradients:
331            return
332
333        # Calculate new gradient.
334        self.gradient = 1000
335        lowest_gradient = int(np.array(valid_gradients).min())
336        if self.gradient is None:
337            # set to lowest + 1
338            self.gradient = lowest_gradient + 1
339        elif self.gradient > lowest_gradient + 1:
340            self.gradient = lowest_gradient + 1
341
342        # update the label on object
343        self.label = self.gradient

345    def assemble_shape(self):
346
347        # Can't assemble the shape if we're not localised.
348        if self.local_pos is None:
349            return False
350
351        # There will never be a shape in the bottom left quadrant.
352        if self.local_pos[0] < 0 and self.local_pos[1] < 0:
353            return
354
355        # Stopping Conditions.
356
357        # Condition 1: We are about to edge follow around a stationary robot that has a larger gradient value than our own.
358        orb_agent = self.get_orbit_object()
359        if orb_agent is None:
360            return False
361
362        inside_shape = self.is_inside_shape()
363
364        if (
365            orb_agent is not None
366            and orb_agent.state == AgentState.LOCALISED
367            and inside_shape
368            and self.gradient <= orb_agent.gradient
369            and self.color_flag == orb_agent.color_flag
370            and self.gradient > 1
371        ):
372            return True
373
374        # Condition 2: We are about to exit the shape.
375        if not inside_shape and self.gradient > 1:
376            return True
377
378        return False

380    def check_bridging(self, tolerance=10):
381        """
382        Checks if self.local_pos lies within a given perpendicular distance (tolerance)
383        from any line segment connecting two COMs, and that the closest point lies
384        between the two COMs (not extended beyond them).
385
386        Args:
387            tolerance (float): Maximum allowed perpendicular distance to consider a bridge.
388
389        Returns:
390            bool: True if bridging condition is met, else False.
391        """
392        # arrays of current position and COMs
393        COMs = np.array(self.shape.centre_of_masses, dtype=np.float64)
394        pos = np.array(self.local_pos, dtype=np.float64)
395        # if there isnt another one, we cant bridge.
396        if len(COMs) < 2:
397            return None
398        # for now lets just consider the closest COM and find all the vectors to other ones
399        deltas = COMs - pos
400        distances = np.linalg.norm(deltas, axis=1)
401        closest_index = np.argmin(distances)
402
403        # go thro and check the other COMs
404        for i in range(len(COMs)):
405            A = COMs[i]
406            B = COMs[closest_index]
407            AB = B - A
408
409            # Skip if the same COMs
410            AB_len_sq = np.dot(AB, AB)
411            if AB_len_sq == 0:
412                continue
413
414            # project the point onto the segment
415            AP = pos - A
416            t = np.dot(AP, AB) / AB_len_sq
417
418            # Check if projection lies within the segment
419            if 0 <= t <= 1:
420                closest_point = A + t * AB
421                # get perpendicular distance, return if in tolerance
422                dist = np.linalg.norm(pos - closest_point)
423                if dist <= tolerance:
424                    return COMs[i]
425
426        return None

428    def assemble_bridge(self):
429        """Bridge to an island if this one is already full!
430        Assumptions: closest COM is the shape we currently belong to."""
431
432        # ain't localised, dont bridge
433        if self.local_pos is None:
434            return
435
436        # We can bridge, lets check if possible
437        desired_COM = self.check_bridging(self.radius)
438        if desired_COM is not None:
439            # below this distance, we encourage bridging. Above, we discourage bridging
440            ideal_COM_distance = 50
441            # what is the average probability we should bridge
442            nominal_bridging_probability = 0.1
443            # calculate
444            dist = np.subtract(desired_COM, self.local_pos)
445            p_bridging = np.clip((ideal_COM_distance / np.linalg.norm(dist)) * nominal_bridging_probability * (self.loops_around-1))
446            #p_bridging = 1
447            print(f"Loops: {self.loops_around}")
448            print(
449                f"probability bridging at {self.local_pos} with distance of {np.linalg.norm(desired_COM)}: {p_bridging}"
450            )
451            if random.random() < p_bridging:
452                # print(f"we have bridged at local position {self.local_pos}, gobally at:{self._pos}")

458        if self.mode == "monochrome":
459            self.color = Colour.white
460

189        if self.is_seed:

198        localised_neighbours = [

210        # Useful for debugging other parts of the agent that rely on localisation to be working properly.

213        #                              seed._pos[1] - self._pos[1]]
214        # return

219        # NOTE: This localisation algorithm is a bit rubbish. It's based off the paper, and was designed to manage
220        # the constraints they were facing with small, low-power, low-compute robots, and asynchronous comms.

243                actual_vec = v * measured_dist

246                new_pos = np.add(agent.local_pos, actual_vec)

259            dist = diff_from_last[0] ** 2 + diff_from_last[1] ** 2
260            last_pos = pos

262        # Add localisation noise to simulate real life.
263        noise_threshold = 0.00

381    def check_bridging(self, tolerance=10):
382        """
383        Checks if self.local_pos lies within a given perpendicular distance (tolerance)
384        from any line segment connecting two COMs, and that the closest point lies
385        between the two COMs (not extended beyond them).
386
387        Args:
388            tolerance (float): Maximum allowed perpendicular distance to consider a bridge.
389
390        Returns:
391            bool: True if bridging condition is met, else False.
392        """
393        # arrays of current position and COMs
394        COMs = np.array(self.shape.centre_of_masses, dtype=np.float64)
395        pos = np.array(self.local_pos, dtype=np.float64)
396        # if there isnt another one, we cant bridge.
397        if len(COMs) < 2:
398            return None
399        # for now lets just consider the closest COM and find all the vectors to other ones
400        deltas = COMs - pos
401        distances = np.linalg.norm(deltas, axis=1)
402        closest_index = np.argmin(distances)
403
404        # go thro and check the other COMs
405        for i in range(len(COMs)):
406            A = COMs[i]
407            B = COMs[closest_index]
408            AB = B - A
409
410            # Skip if the same COMs
411            AB_len_sq = np.dot(AB, AB)
412            if AB_len_sq == 0:
413                continue
414
415            # project the point onto the segment
416            AP = pos - A
417            t = np.dot(AP, AB) / AB_len_sq
418
419            # Check if projection lies within the segment
420            if 0 <= t <= 1:
421                closest_point = A + t * AB
422                # get perpendicular distance, return if in tolerance
423                dist = np.linalg.norm(pos - closest_point)
424                if dist <= tolerance:
425                    return COMs[i]
426
427        return None

381    def check_bridging(self, tolerance=10):
382        """
383        Checks if self.local_pos lies within a given perpendicular distance (tolerance)
384        from any line segment connecting two COMs, and that the closest point lies
385        between the two COMs (not extended beyond them).
386
387        Args:
388            tolerance (float): Maximum allowed perpendicular distance to consider a bridge.
389
390        Returns:
391            bool: True if bridging condition is met, else False.
392        """
393        # arrays of current position and COMs
394        COMs = np.array(self.shape.centre_of_masses, dtype=np.float64)
395        pos = np.array(self.local_pos, dtype=np.float64)
396        # if there isnt another one, we cant bridge.
397        if len(COMs) < 2:
398            return None
399        # for now lets just consider the closest COM and find all the vectors to other ones
400        deltas = COMs - pos
401        distances = np.linalg.norm(deltas, axis=1)
402        closest_index = np.argmin(distances)
403
404        # go thro and check the other COMs
405        for i in range(len(COMs)):
406            A = COMs[i]
407            B = COMs[closest_index]
408            AB = B - A
409
410            # Skip if the same COMs
411            AB_len_sq = np.dot(AB, AB)
412            if AB_len_sq == 0:
413                continue
414
415            # project the point onto the segment
416            AP = pos - A
417            t = np.dot(AP, AB) / AB_len_sq
418
419            # Check if projection lies within the segment
420            if 0 <= t <= 1:
421                closest_point = A + t * AB
422                # get perpendicular distance, return if in tolerance
423                dist = np.linalg.norm(pos - closest_point)
424                if dist <= tolerance:
425                    return COMs[i]
426
427        return None
